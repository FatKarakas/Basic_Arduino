#define trig 7
#define echo 8
#define buzzer 9
float sure,mesafe;
void setup()
{
  pinMode(trig, OUTPUT);
  pinMode(echo,INPUT);
  pinMode(buzzer,OUTPUT);
  Serial.begin(9600);
  digitalWrite(trig,LOW);
}

void loop()
{
  digitalWrite(trig,HIGH);
  delayMicroseconds(10);
  digitalWrite(trig,LOW);
  sure=pulseIn(echo,HIGH);
  mesafe=sure*0.03432/2;
  Serial.println(mesafe);
  if(mesafe<250)
  {
    tone(buzzer,220,50);
    delay(500);
    noTone(buzzer);
  }
}
